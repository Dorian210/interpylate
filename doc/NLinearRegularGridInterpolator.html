 98    Linear grid interpolator : interpolate a 1D array between the indices using a linear (afine) method :
 99    F(x) = a + bx
100    """
101    def __init__(self):
102        """
103        Create the interpolator.
104        """
105        pass

107    def _get_corners(self, vector, i):
108        i0, i1 = i, i + 1
109        l = vector[i0]
110        m = vector[i1]
111        return (l, m)

113    def _make_coefs(self, vector, i):
114        l, m = self._get_corners(vector, i)
115        a = l
116        b = m - a
117        return [a, b]

122        max_ind = size - 2
123        inds[inds>max_ind] = max_ind

336class TriLinearRegularGridInterpolator:
337    """
338    Tri-linear grid interpolator : interpolate a 3D array between the indices using a tri-linear method :
339    F(x, y, z) = a + bx + cy + dz + exy + fxz + gyz + hxyz
340    """
341    def __init__(self):
342        """
343        Create the interpolator.
344        """
345        pass

347    def _get_corners(self, volume, i, j, k):
348        i0, j0, k0, i1, j1, k1 = i, j, k, i + 1, j + 1, k + 1
349        l = volume[i0, j0, k0]
350        m = volume[i1, j0, k0]
351        n = volume[i0, j1, k0]
352        o = volume[i0, j0, k1]
353        p = volume[i1, j1, k0]
354        q = volume[i1, j0, k1]
355        r = volume[i0, j1, k1]
356        s = volume[i1, j1, k1]
357        return (l, m, n, o, p, q, r, s)

359    def _make_coefs(self, volume, i, j, k):
360        l, m, n, o, p, q, r, s = self._get_corners(volume, i, j, k)
361        a = l
362        b = m - a
363        c = n - a
364        d = o - a
365        e = p - a - b - c
366        f = q - a - b - d
367        g = r - a - c - d
368        h = s - a - b - c - d - e - f - g
369        return [a, b, c, d, e, f, g, h]

476class NLinearRegularGridInterpolatorLarge:
477    """
478    N-linear grid interpolator : interpolate a ND array between the indices using a N-linear method.
479    For N<4, consider using ``TriLinearRegularGridInterpolator``, ``BiLinearRegularGridInterpolator``, 
480    or ``LinearRegularGridInterpolator``.
481    For exemple, for a 2D array, the interpolator finds a, b, c and d on each square of the grid such that 
482    F(x, y) = a + bx + cy + dxy
483    """
484    def __init__(self, dim):
485        """
486        Create and initialize the interpolator.
487
488        Parameters
489        ----------
490        dim : int
491            The dimension of the array to be interpolated.
492        """
493        self.dim = dim
494        self.nb_coefs = 2**self.dim
495        self.masks = self._make_masks()
496        self.mat = self._make_mat()
497    
498    def _make_masks(self):
499        masks = np.zeros((self.nb_coefs, self.dim), dtype='bool')
500        for i in range(self.nb_coefs):
501            tmp = np.array(list(bin(i)[2:]), dtype='int')
502            masks[i, (self.dim - tmp.size):] = tmp
503        return masks
504    
505    def _make_mat(self):
506        M = np.logical_not(np.logical_and(self.masks[None, :, :], self.masks[:, None, :]).any(axis=-1)).astype('int')
507        mat = np.linalg.inv(M).astype('int')
508        return mat
509    
510    def _make_corners_inds(self, inds):
511        inds_inds_plus_one = np.repeat(inds[None], 2, axis=0)
512        inds_inds_plus_one[1] += 1
513        arr = np.repeat(np.arange(self.dim)[:, None], self.nb_coefs, axis=1)
514        corners_inds = inds_inds_plus_one[self.masks.T.astype('int'), arr]
515        return corners_inds
516    
517    def _get_corners(self, NDarray, inds):
518        corners_inds = self._make_corners_inds(inds)
519        corners = NDarray[tuple(corners_inds)]
520        return corners
521    
522    def _make_coefs(self, NDarray, inds):
523        corners = self._get_corners(NDarray, inds)
524        coefs = self.mat @ corners
525        return coefs
526    
527    def _get_inds_coords(self, shape, continuous_inds):
528        inds = continuous_inds.astype('int')
529        inds[inds<0] = 0
530        for axis in range(self.dim):
531            max_size = shape[axis] - 2
532            inds[axis, inds[axis]>max_size] = max_size
533        coords = continuous_inds - inds
534        return inds, coords
535    
536    def evaluate(self, NDarray, continuous_inds):
537        """
538        Evaluate the interpolation at the coordinates given as input.
539
540        Parameters
541        ----------
542        NDarray : numpy.ndarray
543            The array to interpolate.
544        continuous_inds : numpy.ndarray of float
545            Coordinates where the interpolation is computed.
546            It's shape should be (dim, n) if dim is the number of dimensions of the array interpolated.
547
548        Returns
549        -------
550        numpy.ndarray of float
551            Interpolated values at the coordinates given.
552            If ``continuous_inds`` is of shape (dim, n), the output will be of shape (n,).
553        """
554        inds, coords = self._get_inds_coords(NDarray.shape, continuous_inds)
555        coefs = self._make_coefs(NDarray, inds)
556        evaluated = _evaluate(coords, self.masks, coefs)
557        return evaluated
558    
559    def _change_masks_and_coef_inds_by_deriving(self, masks, coef_inds, k):
560        to_keep = masks[:, k]
561        new_masks = masks[to_keep]
562        new_masks[:, k] = False
563        new_coef_inds = coef_inds[to_keep]
564        return new_masks, new_coef_inds
565    
566    def grad(self, NDarray, continuous_inds, evaluate_too=False):
567        """
568        Compute the gradient of the interpolated array.
569
570        Parameters
571        ----------
572        NDarray : numpy.ndarray
573            The array to interpolate.
574        continuous_inds : numpy.ndarray of float
575            Coordinates where the gradient is computed.
576            It's shape should be (dim, n) if dim is the number of dimensions of the array interpolated.
577        evaluate_too : bool, optional
578            Set to True to evaluate on these points too, by default False
579
580        Returns
581        -------
582        list of numpy.ndarray of float
583            The derivative of the interpolated array in each axis's direction.
584            If ``continuous_inds`` is of shape (dim, n), each of the output will be of shape (n,).
585        """
586        inds, coords = self._get_inds_coords(NDarray.shape, continuous_inds)
587        coefs = self._make_coefs(NDarray, inds)
588        coef_inds = np.arange(self.nb_coefs)
589        masks = self.masks
590        grad = [None]*self.dim
591        for axis in range(self.dim):
592            new_masks, new_coef_inds = self._change_masks_and_coef_inds_by_deriving(masks, coef_inds, axis)
593            grad[axis] = _evaluate(coords, new_masks, coefs[new_coef_inds])
594        if evaluate_too:
595            evaluated = _evaluate(coords, self.masks, coefs)
596            return grad, evaluated
597        return grad
